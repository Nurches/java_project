package users {
  abstract class User {
    -id: String
    -login: String
    -password: String
    -name: String
    -email: String
    -role: UserRole
    +authenticate(login: String, password: String): boolean
    +getRole(): UserRole
    +toString(): String
    +equals(obj: Object): boolean
    +hashCode(): int
  }

  abstract class Employee {
    -salary: double
    -department: String
    +sendMessage(receiver: Employee, text: String): void
  }

  class Admin {
    +addUser(user: User): void
    +removeUser(user: User): void
    +updateUser(user: User): void
  }

  class Student {
    -major: String
    -yearOfStudy: int
    -gpa: double
    -credits: int
    -courses: List<Course>
    -marks: List<Mark>
    +registerForCourse(course: Course): void
    +viewCourses(): List<Course>
    +viewMarks(): List<Mark>
    +getTranscript(): Transcript
    +calculateGpa(): double
  }

  class BachelorStudent

  class GraduateStudent {
    -supervisor: Researcher
    -diplomaProject: DiplomaProject
    +setSupervisor(supervisor: Researcher): void
  }

  class MasterStudent {
    -researchPapers: List<ResearchPaper>
    +getResearchPapers(): List<ResearchPaper>
    +addResearchPaper(paper: ResearchPaper): void
    +printPapers(comparator: Comparator<ResearchPaper>): void
    +calculateHIndex(): int
    +getTotalCitations(): int
  }

  class PhDStudent {
    -researchPapers: List<ResearchPaper>
    +getResearchPapers(): List<ResearchPaper>
    +addResearchPaper(paper: ResearchPaper): void
    +printPapers(comparator: Comparator<ResearchPaper>): void
    +calculateHIndex(): int
    +getTotalCitations(): int
  }

  class Teacher {
    -title: TeacherTitle
    -courses: List<Course>
    +putMark(student: Student, course: Course, mark: Mark): void
    +sendComplaint(student: Student, text: String, urgency: UrgencyLevel): Complaint
    +assignCourse(course: Course): void
  }

  class Manager {
    -managerType: ManagerType
    +approveRegistration(registration: CourseRegistration): void
    +assignTeacherToCourse(teacher: Teacher, course: Course): void
    +generateReport(): String
  }
}

package research {
  interface Researcher {
    +getResearchPapers(): List<ResearchPaper>
    +addResearchPaper(paper: ResearchPaper): void
    +printPapers(comparator: Comparator<ResearchPaper>): void
    +calculateHIndex(): int
    +getTotalCitations(): int
  }

  class ResearchPaper {
    -title: String
    -authors: List<String>
    -journal: String
    -pages: int
    -citations: int
    -doi: String
    -publishedDate: LocalDate
    +getCitation(format: CitationFormat): String
    +compareTo(other: ResearchPaper): int
    +toString(): String
  }

  class ResearchProject {
    -topic: String
    -publishedPapers: List<ResearchPaper>
    -participants: List<Researcher>
    +addParticipant(participant: Object): void
    +addPaper(paper: ResearchPaper): void
  }

  class ResearchJournal {
    -name: String
    -subscribers: List<User>
    -papers: List<ResearchPaper>
    +subscribe(user: User): void
    +unsubscribe(user: User): void
    +publishPaper(paper: ResearchPaper): void
    +notifySubscribers(paper: ResearchPaper): void
  }

  class DiplomaProject {
    -title: String
    -description: String
  }
}

MasterStudent ..|> Researcher
