package users {
  abstract class User {
    -id: String
    -login: String
    -password: String

    -salary: double
    -department: String
    +sendMessage(receiver: Employee, text): Message

  class Admin

    -major: String
    -yearOfStudy: int
    -gpa: double
    -credits: int
    -failedCoursesCount: int

    +becomeResearcher(): void
    +leaveResearcherRole(): void

@startuml SequenceRegistration
title Sequence - Course Registration (request + approve)
actor Student
actor Manager
participant "StudentMenu" as SUI
participant "ManagerMenu" as MUI
participant "RegistrationService" as RS
participant "RbacService" as RBAC
participant "UniversitySystem" as US
participant "Manager" as M

Student -> SUI : Request registration (courseId)
SUI -> RS : createRequest(student, courseId)
RS -> RBAC : requireRole(STUDENT, BACHELOR_*, MASTER_*, PHD_*)
RS -> US : addRegistration(PENDING)
RS --> SUI : CourseRegistration created

Manager -> MUI : Approve registration
MUI -> RS : approve(manager, registration)
RS -> M : approveRegistration(reg)
M -> M : check eligibility, failed limit
M -> US : registration.approve()
RS -> Student : registerForCourse(course)
RS --> MUI : Approved

@enduml


@startuml SequenceCitationFormat
title Sequence - Export Citation (select paper + format)
actor Researcher as "Student / Teacher / Manager\n(researcherActive)"
participant "ResearchMenuHelper" as RMH
participant "CliIO" as IO
participant "ResearchService" as RS
participant "ResearchPaper" as RP

Researcher -> RMH : Menu: Citation format
RMH -> RP : getResearchPapers()
alt no papers
  RMH --> Researcher : "No papers."
else has papers
  RMH -> IO : chooseFromList(titles)
  IO --> RMH : selected ResearchPaper
  RMH -> Researcher : read format (PLAIN/APA/MLA)
  RMH -> RS : citation(paper, format)
  RS -> RP : getCitation(format)
  RP --> RS : formatted string
  RS --> RMH : citation text
  RMH --> Researcher : print citation
end

@enduml
